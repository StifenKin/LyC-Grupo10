    a, c, d : Float

#+--------------------- Funcion WRITE ---------------------+#

#+--------------------- Asignaciones ---------------------+#
#+--- Asignaciones validas con constantes numericas ---+#
a := 99999.99
a := 99.
a := .9999
a := 32767

#+--- Asignacion con entero negativo ---+#
variable1 := -32768

#+--- ERROR: Constante entera fuera de rango (16 bits signed) ---+#
#+ variable1 := 65536  -- Error: excede rango -32768..32767 +#
#+ variable1 := -32769  -- Error: por debajo del rango minimo +#

#+--- Asignaciones de strings ---+#

#+--- ERROR: Tipos incompatibles en asignacion ---+#
#+ variable1 := "texto"  -- Error: no se puede asignar String a Int +#

#+--------------------- Expresiones Aritmeticas ---------------------+#
#+--- Operaciones basicas ---+#
c := a + 5.5
d := variable1 * 2
a := d / 3.0
variable1 := contador - 10

#+--- Expresiones complejas ---+#
c := -(5.5 + a)
d := (a + variable1) * (c - 2.5)
a := variable1 + c * d / 2.0

#+--- Operador unario ---+#
c := -a
variable1 := -contador

#+--- ERROR: Tipos incompatibles en operaciones ---+#
#+ c := a + "texto"  -- Error: no se puede sumar Float + String +#
#+ variable1 := -b  -- Error: operador unario en String +#

#+--------------------- Estructuras de Control ---------------------+#
#+--- Condicionales simples ---+#
if (variable1 > 0) {
    write("Positivo")
}

if (a >= 10.5) {
    variable1 := variable1 + 1
} else {
    variable1 := variable1 - 1
}

#+--- Condiciones complejas ---+#
if (a > 0.0 AND variable1 < 100) {
    write("Condicion AND")
}

if (a <= 5.0 OR contador >= 10) {
    write("Condicion OR")
}

if (NOT (variable1 == 0)) {
    write("Condicion NOT")
}

#+--- Ciclos ---+#
while (contador < 10) {
    contador := contador + 1
    write("Iteracion")
}

#+--- Ciclos anidados ---+#
while (variable1 > 0) {
    if (variable1 % 2 == 0) {
        variable1 := variable1 / 2
    } else {
        variable1 := variable1 * 3 + 1
    }
}

if (isZero(variable1)) {
    write("Variable es cero")
}

if (isZero(a - 5.0)) {
    write("Expresion es cero")
}
